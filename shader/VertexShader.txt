#version 330 core
layout (location = 0) in vec3 position;
layout (location = 1) in vec3 colorin;

out vec4 colori;

uniform mat4 transform;
uniform mat4 model;
uniform mat4 view;
uniform mat4 projection;

void main()
{
    gl_Position = projection * view * model * vec4(position, 1.0f);
    colori = vec4(colorin, 1.0f);
};
